```SQL

CREATE TABLE U_LIST
(
    ID       VARCHAR2(20)    NOT NULL, 
    PWD      VARCHAR2(20)    NULL, 
    NAME     VARCHAR2(20)    NULL, 
    UMODE    NUMBER          NULL, 
    CONSTRAINT U_LIST_PK PRIMARY KEY (ID)
)

CREATE TABLE PLAYER
(
    ID          NUMBER          NOT NULL, 
    PIC         VARCHAR2(20)    NULL, 
    NAME        VARCHAR2(20)    NULL, 
    POSITION    VARCHAR2(20)    NULL, 
    TEAM        VARCHAR2(20)    NULL, 
    CONSTRAINT PLAYER_PK PRIMARY KEY (ID)
)

CREATE TABLE P_HISTORY
(
    TEAM       VARCHAR2(20)    NULL, 
    ERA        NUMBER          NULL, 
    GAME       NUMBER          NULL, 
    AVG        NUMBER          NULL, 
    HR         NUMBER          NULL, 
    YEAR       NUMBER          NULL, 
    WPCT       NUMBER          NULL, 
    BACKNUM    NUMBER          NULL, 
    ID         NUMBER          NULL, 
    CONSTRAINT P_HISTORY_PK PRIMARY KEY (YEAR, ID)
)

CREATE TABLE TEAM
(
    TNAME       VARCHAR2(20)    NULL, 
    HSTA        VARCHAR2(40)    NULL, 
    DIRECTOR    VARCHAR2(20)    NULL, 
    ID          NUMBER          NULL, 
    VICTORY     NUMBER          NULL, 
    CONSTRAINT TEAM_PK PRIMARY KEY (ID)
)

CREATE TABLE T_HISTORY
(
    YEAR     NUMBER          NULL, 
    TEAM     VARCHAR2(20)    NULL, 
    GRADE    NUMBER          NULL, 
    ID       NUMBER          NULL, 
    WGAME    NUMBER          NULL, 
    LGAME    NUMBER          NULL, 
    TGAME    NUMBER          NULL, 
    CONSTRAINT T_HISTORY_PK PRIMARY KEY (YEAR, ID)
)


ALTER TABLE P_HISTORY ADD FOREIGN KEY (ID) REFERENCES PLAYER (ID)
ALTER TABLE T_HISTORY ADD FOREIGN KEY (ID) REFERENCES TEAM (ID)

CREATE SEQUENCE P_ID INCREMENT BY 1 START WITH 101
CREATE SEQUENCE T_ID INCREMENT BY 1 START WITH 1

```
